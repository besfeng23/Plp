# OTA Channel Sync Schema Foundation

This repository does not currently have a dedicated migrations directory or an applied migration workflow. The schema below is therefore a **database design document only** and must not be treated as applied to Supabase until it is converted into a reviewed migration and executed in the target environment.

No live OTA credentials, tokens, or API secrets should be stored in these tables in plain text. Future credential storage should use an approved secrets manager or encrypted server-side storage outside client-visible payloads.

## Shared conventions

- Primary keys should use UUIDs generated by Postgres/Supabase.
- `created_at` should default to `now()`.
- `updated_at` should be maintained by trigger or explicit server-side updates where records are mutable.
- Channel IDs should use the shared constants documented in `lib/ota/channels.js`: `booking_com`, `agoda`, `airbnb`, `expedia`, `vrbo`, `direct`, and `manual`.
- Status values should align with: `imported`, `mapped`, `unmapped`, `blocked`, `conflict`, `cancelled`, `modified`, `needs_review`, `synced`, and `failed`.
- `raw_payload` JSONB columns are for provider payload snapshots, iCal event data, and diagnostic metadata. Do not place plaintext credentials in `raw_payload`.
- `channel_key` is intentionally repeated on operational tables even when `channel_id` exists. This denormalized value is for audit stability, simpler support queries, and readable raw import review if an upstream channel row is later renamed or retired. Migrations should still keep it aligned with `ota_channels.channel_key` through server-side validation where practical.

## Proposed SQL

```sql
create table if not exists ota_channels (
  id uuid primary key default gen_random_uuid(),
  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now(),
  channel_key text not null unique,
  display_name text not null,
  channel_type text not null check (channel_type in ('ota', 'ical', 'direct', 'manual', 'channel_manager')),
  status text not null default 'needs_review',
  supports_import boolean not null default false,
  supports_export boolean not null default false,
  notes text,
  raw_payload jsonb not null default '{}'::jsonb
);

create table if not exists ota_channel_connections (
  id uuid primary key default gen_random_uuid(),
  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now(),
  channel_id uuid references ota_channels(id) on delete cascade,
  channel_key text not null,
  connection_name text not null,
  connection_status text not null default 'needs_review',
  auth_mode text not null default 'not_connected',
  credential_reference text,
  external_account_id text,
  last_sync_at timestamptz,
  review_status text not null default 'needs_review',
  raw_payload jsonb not null default '{}'::jsonb,
  constraint ota_connections_no_plain_secret check (credential_reference is null or credential_reference !~* '(password|secret|token|key)=')
);

create table if not exists ota_room_mappings (
  id uuid primary key default gen_random_uuid(),
  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now(),
  channel_id uuid references ota_channels(id) on delete cascade,
  connection_id uuid references ota_channel_connections(id) on delete cascade,
  channel_key text not null,
  external_room_id text not null,
  external_room_name text,
  accommodation_id uuid,
  accommodation_name text,
  mapping_status text not null default 'unmapped',
  review_status text not null default 'needs_review',
  raw_payload jsonb not null default '{}'::jsonb,
  unique (channel_key, external_room_id)
);

create table if not exists ota_rate_mappings (
  id uuid primary key default gen_random_uuid(),
  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now(),
  room_mapping_id uuid references ota_room_mappings(id) on delete cascade,
  channel_key text not null,
  external_rate_id text not null,
  external_rate_name text,
  local_rate_code text,
  currency text,
  mapping_status text not null default 'unmapped',
  review_status text not null default 'needs_review',
  raw_payload jsonb not null default '{}'::jsonb,
  unique (channel_key, external_rate_id)
);

create table if not exists external_bookings (
  id uuid primary key default gen_random_uuid(),
  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now(),
  channel_id uuid references ota_channels(id),
  connection_id uuid references ota_channel_connections(id),
  channel_key text not null,
  external_booking_id text not null,
  external_room_id text,
  booking_reference text,
  guest_display_name text,
  guest_contact_masked text,
  check_in date,
  check_out date,
  guest_count integer,
  external_status text,
  import_status text not null default 'imported',
  review_status text not null default 'needs_review',
  conflict_status text not null default 'needs_review',
  raw_payload jsonb not null default '{}'::jsonb,
  constraint external_bookings_valid_stay check (check_in is null or check_out is null or check_out > check_in),
  constraint external_bookings_positive_guests check (guest_count is null or guest_count > 0),
  unique (channel_key, external_booking_id)
);

create table if not exists external_booking_events (
  id uuid primary key default gen_random_uuid(),
  created_at timestamptz not null default now(),
  external_booking_id uuid references external_bookings(id) on delete cascade,
  channel_key text not null,
  event_type text not null,
  event_status text not null default 'imported',
  occurred_at timestamptz,
  processed_at timestamptz,
  review_status text not null default 'needs_review',
  raw_payload jsonb not null default '{}'::jsonb
);

create table if not exists calendar_feeds (
  id uuid primary key default gen_random_uuid(),
  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now(),
  channel_id uuid references ota_channels(id),
  connection_id uuid references ota_channel_connections(id),
  channel_key text not null,
  feed_direction text not null check (feed_direction in ('import', 'export')),
  feed_name text not null,
  feed_url_reference text,
  feed_status text not null default 'needs_review',
  last_synced_at timestamptz,
  review_status text not null default 'needs_review',
  raw_payload jsonb not null default '{}'::jsonb
);

create table if not exists calendar_sync_runs (
  id uuid primary key default gen_random_uuid(),
  created_at timestamptz not null default now(),
  calendar_feed_id uuid references calendar_feeds(id) on delete set null,
  channel_key text not null,
  run_type text not null check (run_type in ('import', 'export', 'dry_run')),
  run_status text not null default 'needs_review',
  started_at timestamptz,
  finished_at timestamptz,
  imported_count integer not null default 0,
  blocked_count integer not null default 0,
  conflict_count integer not null default 0,
  error_message text,
  raw_payload jsonb not null default '{}'::jsonb,
  constraint calendar_sync_runs_valid_duration check (started_at is null or finished_at is null or finished_at >= started_at)
);

create table if not exists availability_blocks (
  id uuid primary key default gen_random_uuid(),
  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now(),
  external_booking_id uuid references external_bookings(id) on delete set null,
  channel_key text not null,
  accommodation_id uuid,
  accommodation_name text,
  block_start date not null,
  block_end date not null,
  block_status text not null default 'blocked',
  block_reason text not null default 'external_booking_import',
  review_status text not null default 'needs_review',
  raw_payload jsonb not null default '{}'::jsonb,
  constraint availability_blocks_valid_range check (block_end > block_start)
);

create table if not exists availability_conflicts (
  id uuid primary key default gen_random_uuid(),
  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now(),
  external_booking_id uuid references external_bookings(id) on delete set null,
  availability_block_id uuid references availability_blocks(id) on delete set null,
  channel_key text not null,
  conflict_type text not null,
  conflict_status text not null default 'conflict',
  severity text not null default 'needs_review',
  detected_at timestamptz not null default now(),
  resolved_at timestamptz,
  resolved_by text,
  resolution_notes text,
  review_status text not null default 'needs_review',
  raw_payload jsonb not null default '{}'::jsonb,
  constraint availability_conflicts_valid_resolution check (resolved_at is null or resolved_at >= detected_at)
);
```

## Implementation note

Before applying this design, add foreign keys to the actual PLP accommodation/booking table IDs after confirming the production schema names and row-level-security posture. The Phase 2A UI is intentionally read-only placeholder foundation work.
